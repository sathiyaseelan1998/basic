interface Inter
{
	void a();

}

interface Inter1 extends Inter
{
	void b();

}

class Abc implements Inter1
{

	void a()
	{
	
		System.out.println("method a");
	}
	void b()
	{
	
		System.out.println("method b");
	}

	public static void main(String args[])
	{
		Aa ob=new Aa();
		ob.a();
		ob.b();

	}


}
